% \modeldef
%
% Used to define a model element within a table. This command can only
% be used in the \tabular environment.
%
%   #1 - image of the model element
%   #2 - name of the model element
%   #3 - text describing the model element
%
\newcommand{\modeldef}[3]{\includegraphics{#1} & \textbf{#2} -- #3}
